Reason
Display a visual indication (busy cursor) when long operations are taking place.
Code Changes
Change COM.footprint.ApplicationCatalog.view.ApplicationCatalogView launchApplicationView() to comment out the hide() and show() calls as indicated below:
/**
* Launch the selected application.
*/
private void launchApplication ( ) {

setCursor(new java.awt.Cursor(java.awt.Cursor.WAIT_CURSOR));
Action anAction = (Action)getcontainer().getSelectedNode().getObject();
String appName = getcontainer().getSelectedNode().getLabel();
java.awt.event.ActionEvent e = new java.awt.event.ActionEvent(getParentFrame(),java.awt.event.ActionEvent.ACTION_FIRST,"");
if (anAction.isEnabled()) {
//getParent().hide();
try {
anAction.actionPerformed(e);
disposeParent();
setCursor(new java.awt.Cursor(java.awt.Cursor.DEFAULT_CURSOR));
}
catch(Exception x) {
setCursor(new java.awt.Cursor(java.awt.Cursor.DEFAULT_CURSOR));
handleException(new ApplicationLaunchFailedException(x, appName));
// getParent().show();
}
}
else {
//getParent().hide();
setCursor(new java.awt.Cursor(java.awt.Cursor.DEFAULT_CURSOR));
handleException(new ApplicationLaunchFailedException(appName));
//getParent().show();
}
}
