package COM.footprint.telecom.model;

/* Copyright (c) 1999 IBM Corporation. All rights reserved. */

/**
* The TelecomProvider is the interface that a class implements to provide
* telecom functionality to the Contact Center framework. It provides methods
* to start and stop the provider functionality, and maintains state
* information, notifying any interested objects of changes in state.
*
* @exception COM.footprint.telecom.model.TelecomProviderException
*/
public interface TelecomProvider
{
/**
* Indicator specifying that the telecom provider is in service.
*/
public static final int IN_SERVICE = 0;
/**
* Indicator specifying that the telecom provider is out of service.
*/
public static final int OUT_OF_SERVICE = 1;
/**
* Indicator specifying that the telecom provider is not started.
*/
public static final int SHUTDOWN = 2;
/**
* Indicator specifying that the telecom provider is not active.
*/
public static final int DISCONNECTED = 3;
/**
* Indicator specifying that the telecom provider is in an unknown state.
*/
public static final int UNKNOWN = 4;

/**
* Registers interest for a specific listener in receiving notification of
* state changes for this telecom provider.
*
* @param listener COM.footprint.telecom.model.ProviderStateListener, the
* listener object that wishes to start being notified of
* telecom provider state changes
*/
void addProviderStateListener(ProviderStateListener listener);

/**
* Retrieves the current state of the telecom provider.
*
* @return int - the current state value
*/
int getState();

/**
* Retrieves the descriptor for the telecom provider.
*
* @return COM.footprint.telecom.model.TelecomDescriptor - the descriptor
* value for the telecom provider
*/
TelecomDescriptor getTelecomDescriptor();

/**
* Deregisters interest for a specific listener in receiving notification of
* state changes for this telecom provider.
*
* @param listener COM.footprint.telecom.model.ProviderStateListener, the
* listener object that wishes to stop being notified of
* telecom provider state changes
*/
void removeProviderStateListener(ProviderStateListener listener);

/**
* Starts the telecom provider.
*
* @exception COM.footprint.telecom.model.TelecomProviderException
*/
void start()
throws TelecomProviderException;

/**
* Stops the telecom provider.
*
* @exception COM.footprint.telecom.model.TelecomProviderException
*/
void stop()
throws TelecomProviderException;
}

Fragment 2

package COM.footprint.telecom.model;

/* Copyright (c) 1999 IBM Corporation. All rights reserved. */

/**
* The ContactDataChangeListener is the interface that a class implements in
* order to receive notification that a ContactData object has undergone a
* change in state. The listening object registers interest in this
* notification by calling the addContactDataChangeListener() method for the
* ContactData object. The ContactData object notifies listeners of a change
* in state by calling the contactDataChange() method.
*
* @see COM.footprint.telecom.model.ContactData
* @see COM.footprint.telecom.model.ContactData#addContactDataChangeListener
*/
public interface ContactDataChangeListener
{
/**
* Performs the processing required when a ContactData object has changed
* state.
*
* @param event COM.footprint.telecom.model.ContactDataChangeEvent,
* the event describing the changes that have occurred in the
* ContactData object
*
* @see COM.footprint.telecom.model.ContactData
* @see COM.footprint.telecom.model.ContactDataChangeEvent
*/
void contactDataChange(ContactDataChangeEvent event);
}
