<!--==================================================================

    Parameter Entity Declarations

    ==================================================================-->

 <!ENTITY % DTDVersion '"V0.1.1"' >

<!ENTITY % OtherNotationData "RCDATA" >

<!ENTITY % XRefTarget       "

    ID                      ID              #IMPLIED

    XRefText         CDATA          #IMPLIED

" >

<!ENTITY % PhLike           "

    #PCDATA | Address | CatalogRef | Cit | L |

    MMObj | OfferingRef | Ph | Phone | RefKey |

    Q | Term | XPh | XRef

"                                                      -- Phrase-Like -->

 <!--==================================================================

    Document Element

    ==================================================================-->

An element declaration follows:

<!ELEMENT  Catalog           - O (Prolog?, Frontm, Body)

                                            comment --> --- Catalog Document --

>

An attribute list declaration follows:

<!ATTLIST  Catalog

    CatalogID        CDATA         #REQUIRED

    Class               NAME          #IMPLIED

    DTDVersion       CDATA        #FIXED    %DTDVersion;  <--- parameter entity reference

>

 <!--==================================================================

    Document Hierarchy Elements

    ==================================================================-->

 <!ELEMENT  Prolog           - O (PropDefs?, BibEntryDefs?, IdxDefs?) <-- content model

                                                                         -- Document Prolog -->

 <!ELEMENT  FrontM           - O (Edition )             -- Front Matter -->


<!ELEMENT  Edition          O O (#PCDATA)*         -- Edition Title -->


<!-- V0.0.4: Add Glossary to the DTD                               WLW -->

<!ELEMENT  Body                  O O (Conferences | Roadmaps | Courses |

                                              AdditionalCourses | SelfStudyCourses |

                                              OtherOfferings | TermsAndConds | Trademarks |

                                              SchedByCourse | SchedByLocation | Maps |

                                              Glossary | Index | CatalogPart)*

                                             +(%InclExcl;)                               -- Body -->

 <!ATTLIST (FrontM, Body)

    Class            NAME         #IMPLIED

    Style            CDATA        #IMPLIED

>

 <!--==================================================================

    Paragraph-Like Elements

    ==================================================================-->

 <!ELEMENT  Lines            - - ( #PCDATA | Address | MMObj | Ph | Phone |

                                             Q | RefKey | Term | XPh)*  -- Lines -->

<!ATTLIST  Lines

    Obj                ENTITY          #CONREF

    Class              NAME           #IMPLIED

    Style              CDATA          #IMPLIED

    %XRefTarget;

>

                                    <!--============ Multi-Media Object-->

 <!ELEMENT  MMObj            - O (ObjRef?, TextAlt?)

                                                -- Multi-Media Object --

>

<!ATTLIST  MMObj

    Class            NAME         #IMPLIED

    Style            CDATA        #IMPLIED

    Label            CDATA        #IMPLIED

    %XRefTarget;

>

 <!ELEMENT  ObjRef           - O EMPTY             -- Object Reference -->

<!ATTLIST  ObjRef

    Obj                ENTITY         #REQUIRED

    EditScale       CDATA           #IMPLIED

>

 <!ELEMENT  TextAlt          - O (#PCDATA | Annot | DL | Lines | Note |

                                 NoteList | OL | P | Ph | SL | Term |

                                 UL | Xmp)* -(MMObj)       -- Text Alternative -->


<!ELEMENT  P                - O (%PhAndPLike.NoP;)*      -- Paragraph -->

<!ATTLIST  P

    Class            NAME         #IMPLIED

    Style            CDATA        #IMPLIED

    %XRefTarget;

>

 <!--==================================================================

    Phrase-Like Elements

    ==================================================================-->

                                    <!--======================= Address-->

 <!ELEMENT  Address          - O (

                                            #PCDATA |

                                             Term |

                                                           -- Address -->

<!ATTLIST  Address

    Class                      NAME                #IMPLIED

    Style                       CDATA              #IMPLIED

    %XRefTarget;
